{\conta{AG} \Gamma; \Delta_N; \Xi_N; \Gamma_{N1}; \Delta_{N1}; f, \lstack{C}; \lstack{P}; \Omega_N; \Sigma
\rightarrow \outsem}
{
   \begin{gathered}

      \ma{AG} \Gamma; \Delta; \Xi_N; \Gamma_{N1}; \Delta_{N1}; \Xi; \Omega; f', \lstack{C}; \lstack{P}; \Omega_N;
         \Delta_N; \Sigma \rightarrow \outsem
   \end{gathered}
